🚀 快速开始

# 1. 克隆项目
git clone git@github.com:z58362026/amazing.git
cd amazing

# 2. 初始化
make init

# 3. 启动数据库
docker-compose -f docker-compose.dev.yml up -d

# 4. 启动服务
make dev

# 5. 申请角色 (对话式)
python3 scripts/amazing-cli.py role chat "我是前端开发"
